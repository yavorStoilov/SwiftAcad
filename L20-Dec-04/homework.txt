Задача 2 (да си припомним малко и от стария материал):
	Напишете функция, която приема като входен параметър масив от strings
	и подрежда масивът във възходящ ред по дължината на тези strings.

	Примерен резултат:
		["banana", "kiwi", "strawberry", "ice"] =>
		["ice", "kiwi", "banana", "strawberry"]




(function(){
    "use strict"

    // задача 1:

    function showFirst(){
        var headingOne = document.createElement("h1");
        headingOne.textContent = "PAGE DOM LOADED";
        document.body. appendChild(headingOne);
    }

    function thenShow() {
        var headingTwo = document.createElement("h1");
        headingTwo.textContent = "PAGE BOM LOADED";
        document.body.appendChild(headingTwo);
    }

    window.onload = thenShow;

    document.addEventListener("DOMContentLoaded", showFirst);
    document.addEventListener("window.onload",thenShow);

    function clickPosition(event) {
        var el = document.createElement("h2");
        el.innerHTML = "click captured: " + event.pageX + "X, and " + event.pageY + "Y";
        document.body.appendChild(el);
          
      }
      
      document.addEventListener("click", clickPosition);
